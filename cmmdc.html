Problemă:

      Se citesc două fracții. Să se afișeze suma lor.

Rezolvare:


#include < bits/stdc++.h >

using namespace std;
ifstream fin ("date.in");
ofstream fout ("date.out");

int a,b,ca,cb,r;
int main()
{

fin >> a >> b;
ca = a; cb = b;
while ( b != 0 )
{

r = a % b;
a = b;
b = r;

}
fout << ca/a << "/" << cb/a;
return 0;

}
