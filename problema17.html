Problema 17:

       Din fişierul text tablou.in se citesc: un număr natural n (n≤100) reprezentând numărul de linii și numărul de coloane ale unui tablou bidimensional. De pe următoarele linii se citesc elementele tabloului, câte n componente de pe fiecare linie, numere naturale nenule cu cel mult trei cifre fiecare.
      Pe prima linie a fișierului tablou.out, afișați valoarea celui mai mic element din matrice.
       Determinați câte elemente situate sub diagonala secundară (exclusiv diagonala!) au suma cifrelor un număr par și afișați acest număr pe al doilea rând al fișierului tablou.out. Dacă nu există niciun număr cu proprietatea cerută, se va afișa în fișier mesajul Nu există! .

Exemplu:


tablou.in:
4
1 2 1 3
43 454 234 12
23 77 99 1
8 6 5 3
tablou.out:
1
2


Reozolvare:



#include < bits/stdc++.h >

using namespace std;
ifstream fin ("tablou.in");
ofstream fout ("tablou.out");

int a[101][101],n,i,j,mini,copie,s,nr;
int main()
{

fin>>n;
for(i=1; i<=n; i++)

for(j=1; j<=n; j++)

fin>>a[i][j];

mini=999999;
for(i=1; i<=n; i++)

for(j=1; j<=n; j++)

if(a[i][j] < mini)
    mini=a[i][j];

fout << mini << endl;
for(i=1; i<=n; i++)

for(j=1; j<=n; j++)

if(i+j>n+1)

{

s=0;
copie=a[i][j];
while(copie!=0)
{

s=s+copie%10;
copie/=10;

}
if(s%2==0)

nr++;

}

if(nr==0)

fout<<"Nu exista!";

else

fout << nr;


return 0;

}
